Ruby Enumerable Cheatsheet

  • >> Enumerable.instance_methods.sort

    => [:all?, :any?, :chunk, :collect, :collect_concat, :count, :cycle, :detect, :drop, :drop_while, :each_cons, :each_entry, :each_slice, :each_with_index, :each_with_object, :entries, :find, :find_all, :find_index, :first, :flat_map, :grep, :group_by, :include?, :inject, :map, :max, :max_by, :member?, :min, :min_by, :minmax, :minmax_by, :none?, :one?, :partition, :reduce, :reject, :reverse_each, :select, :slice_before, :sort, :sort_by, :take, :take_while, :to_a, :zip]

  • Searching:

    >> [1,2,3,4,5].select { |i| i > 3 }

    => [4,5]

    >> [1,2,3,4,5].detect { |i| i > 3 }

    => 4

    >> [1,2,3,4,5].reject { |i| i > 3 }

    => [1,2,3]

  • Sorting:

    >> [1, "5", 2, "7", "3"].sort_by { |a| a.to_i }

    => [1, 2, "3", "5", "7"]

    >> [1, "5", 2, "7", "3"].sort_by(&:to_i)

    => [1, 2, "3", "5", "7"]

    • Any? and All?:

    >> [2,4,6,8].all?(&:even?)

    => true

    >> [2,4,5,8].any? { |i| i % 2 == 0 }

    => true

    >> [2,4,5,8].all? { |i| i % 2 == 0 }

    => false

  • Reduce:

    >> [1,2,3].reduce(:+)

    => 6

  • Each:

    >> hash = Hash.new

    %w(cat dog wombat).each_with_index { |item, index|

    hash[item] = index } hash #=> {"cat"=>0, "dog"=>1, "wombat"=>2}

  • Include?:

    IO.constants.include? :SEEK_SET #=> true

    IO.constants.include? :SEEK_NO_FURTHER #=> false

  • MAP:

    (1..4).map { |i| i*i } #=> [1, 4, 9, 16]

    (1..4).collect { "cat" } #=> ["cat", "cat", "cat", "cat"]

  • Reverse:

    (1..3).reverse_each { |v| p v }

    produces:

    3 2 1
